Part 1: Vectors (int[])

Aquests mètodes s'enfoquen a crear i processar vectors (arrays d'una dimensió).

/// <summary>
/// Exercici 01
/// Retorna un array de longitud 'n' amb els N primers números naturals.
/// Exemple: n=3 -> [1, 2, 3]
/// </summary>
public static int[] OmpleNaturals(int n);

/// <summary>
/// /// Exercici 02
/// Retorna un array de longitud 'n' amb números enters aleatoris entre -100 i 100.
/// </summary>
public static int[] GenerarTaula(int n);

/// <summary>
/// /// Exercici 03
/// Retorna la suma de tots els elements de la 'taula'.
/// </summary>
public static int SumarTaula(int[] taula);

/// <summary>
/// /// Exercici 04
/// Retorna el valor més petit de la 'taula'.
/// </summary>
public static int MinimTaula(int[] taula);

/// <summary>
/// /// Exercici 05
/// Retorna l'índex (posició) del valor més petit de la 'taula'.
/// Si hi ha empats, retorna la primera posició.
/// </summary>
public static int PosicioMinimTaula(int[] taula);

/// <summary>
/// /// Exercici 06
/// Retorna el valor més gran de la 'taula'.
/// </summary>
public static int MaximTaula(int[] taula);

/// <summary>
/// /// Exercici 07
/// Retorna el promig de tots els elements de la 'taula'.
/// </summary>
public static double MitjanaTaula(int[] taula);

/// <summary>
/// /// Exercici 08
/// MODIFICA la 'taula' original, invertint l'ordre dels seus elements.
/// (El primer element passa a ser l'últim, el segon el penúltim, etc.)
/// </summary>
public static void CapgirarTaula(int[] taula);

/// <summary>
/// /// Exercici 09
/// Retorna la primera posició (índex) on apareix 'valor' a la taula 't'.
/// Si 'valor' no apareix, retorna -1.
/// </summary>
public static int PosicioDe(int[] t, int valor);

/// <summary>
/// /// Exercici 10
/// Retorna 'true' si 'valor' apareix dins la taula 't', cridant a 'PosicioDe'.
/// </summary>
public static bool Conte(int[] t, int valor);

/// <summary>
/// /// Exercici 11
/// Retorna 'true' si 'valor' apareix a la taula 't' dins l'interval
/// d'índexs [indexDesDe, indexFinsA] (ambdós inclosos).
/// </summary>
public static bool ConteEnRang(int[] t, int valor, int indexDesDe, int indexFinsA);

/// <summary>
/// /// Exercici 12
/// Retorna un string amb el contingut de la taula, formatat.
/// Exemple: per a [10, 20] ha de retornar "Index 0: 10\nIndex 1: 20"
/// </summary>
public static string TaulaAString(int[] taula);

/// <summary>
/// /// Exercici 13
/// Donades dues taules ordenades i sense elements repetits, retorna una nova taula amb els elements COMUNS.
/// </summary>
public static int[] ComunsOrdenades(int[] t1, int[] t2);

/// <summary>
/// /// Exercici 14
/// Donades dues taules ordenades i sense elements repetits, retorna una nova taula amb els elements NO COMUNS
/// (elements que estan a t1 però no a t2, i elements de t2 que no estan a t1).
/// </summary>
public static int[] NoComunsOrdenades(int[] t1, int[] t2);

/// <summary>
/// /// Exercici 15
/// Donades dues taules ordenades i sense elements repetits, retorna una nova taula (ordenada)
/// que és la FUSIÓ de les dues, però sense elements repetits
/// </summary>
public static int[] FusioOrdenades(int[] t1, int[] t2);

Part 2: Matrius (int[,])

Aquests mètodes s'enfoquen a crear i processar matrius (arrays de dues dimensions).

/// <summary>
/// Exercici 16
/// Retorna una matriu NxM plena per files.
/// Exemple (n=3, m=2):
/// 1 1
/// 2 2
/// 3 3
/// </summary>
public static int[,] OmplirMatriuPerFiles(int n, int m);

/// <summary>
/// Exercici 17
/// Retorna una matriu NxM plena per columnes.
/// Exemple (n=3, m=2):
/// 1 2
/// 1 2
/// 1 2
/// </summary>
public static int[,] OmplirMatriuPerColumnes(int n, int m);

/// <summary>
/// Exercici 18
/// Retorna una matriu NxM plena amb números naturals consecutius (començant per 1).
/// Exemple (n=2, m=3):
/// 1 2 3
/// 4 5 6
/// </summary>
public static int[,] OmplirMatriuConsecutius(int n, int m);

/// <summary>
/// Exercici 19
/// Retorna una matriu NxM plena amb números naturals consecutius en ordre invers.
/// Exemple (n=2, m=3) -> Total 6 elements:
/// 6 5 4
/// 3 2 1
/// </summary>
public static int[,] OmplirMatriuConsecutiusInvers(int n, int m);

/// <summary>
/// Exercici 20
/// Retorna una matriu quadrada NxN que és la matriu identitat.
/// (1 a la diagonal, 0 a la resta).
/// </summary>
public static int[,] OmplirMatriuIdentitat(int n);

/// <summary>
/// Exercici 21
/// Retorna el valor més petit dins de la 'matriu'.
/// </summary>
public static int MinimMatriu(int[,] matriu);

/// <summary>
/// Exercici 22
/// Retorna un array de 2 enters [fila, columna] amb la posició
/// del valor més petit de la 'matriu'.
/// </summary>
public static int[] PosicioMinimMatriu(int[,] matriu);
